**check_db() → dict** (async):
```python
async def check_db() -> dict:
    """Verify DB connectivity by executing SELECT 1."""
    engine = get_engine()
    if engine is None:
        return {"status": "unavailable", "detail": "engine not initialised"}
    try:
        async with get_async_session() as session:
            result = await asyncio.wait_for(
                session.execute(text("SELECT 1")),
                timeout=settings.HEALTH_CHECK_TIMEOUT,
            )
            result.scalar()
        return {"status": "ok"}
    except Exception as exc:
        return {"status": "error", "detail": str(exc)}
```

**check_kafka() → dict** (sync):
```python
def check_kafka() -> dict:
    """Verify Kafka broker reachability via list_topics."""
    try:
        producer = Producer({
            "bootstrap.servers": settings.KAFKA_BOOTSTRAP_SERVERS,
            "socket.timeout.ms": settings.HEALTH_CHECK_TIMEOUT * 1000,
        })
        metadata = producer.list_topics(timeout=settings.HEALTH_CHECK_TIMEOUT)
        broker_count = len(metadata.brokers)
        del producer
        return {"status": "ok", "brokers": broker_count}
    except Exception as exc:
        return {"status": "error", "detail": str(exc)}
```

**check_model_freshness() → dict** (async):
```python
async def check_model_freshness() -> dict:
    """Check if the active model was trained within HEALTH_MODEL_FRESHNESS_DAYS."""
    engine = get_engine()
    if engine is None:
        return {"status": "unavailable", "detail": "engine not initialised"}
    try:
        async with get_async_session() as session:
            result = await session.execute(
                text(
                    "SELECT trained_at FROM model_registry "
                    "WHERE is_active = true "
                    "ORDER BY trained_at DESC LIMIT 1"
                )
            )
            row = result.fetchone()
        if row is None:
            return {"status": "unavailable", "detail": "no active model"}
        trained_at = row[0]
        age = datetime.now(timezone.utc) - trained_at
        status = "ok" if age.days <= settings.HEALTH_MODEL_FRESHNESS_DAYS else "stale"
        result_dict = {
            "status": status,
            "model_age_days": age.days,
            "trained_at": trained_at.isoformat(),
        }
        if status == "stale":
            result_dict["threshold_days"] = settings.HEALTH_MODEL_FRESHNESS_DAYS
        return result_dict
    except Exception as exc:
        return {"status": "error", "detail": str(exc)}
```

**check_prediction_staleness() → dict** (async):

```python
async def check_prediction_staleness() -> dict:
    """Check if predictions are fresher than HEALTH_PREDICTION_STALENESS_HOURS."""
    engine = get_engine()
    if engine is None:
        return {"status": "unavailable", "detail": "engine not initialised"}
    try:
        async with get_async_session() as session:
            result = await session.execute(
                text("SELECT MAX(created_at) FROM predictions")
            )
            row = result.fetchone()
        if row is None or row[0] is None:
            return {"status": "unavailable", "detail": "no predictions found"}
        last_at = row[0]
        age = datetime.now(timezone.utc) - last_at
        age_hours = round(age.total_seconds() / 3600, 1)
        status = "ok" if age_hours <= settings.HEALTH_PREDICTION_STALENESS_HOURS else "stale"
        result_dict = {
            "status": status,
            "last_prediction_age_hours": age_hours,
            "last_prediction_at": last_at.isoformat(),
        }
        if status == "stale":
            result_dict["threshold_hours"] = settings.HEALTH_PREDICTION_STALENESS_HOURS
        return result_dict
    except Exception as exc:
        return {"status": "error", "detail": str(exc)}
```

**run_deep_checks() → dict** (async):
```python
async def run_deep_checks() -> dict:
    """Run all health checks concurrently and compute overall status."""
    loop = asyncio.get_event_loop()

    db_check, model_check, pred_check, kafka_check = await asyncio.gather(
        check_db(),
        check_model_freshness(),
        check_prediction_staleness(),
        loop.run_in_executor(None, check_kafka),
    )

    checks = {
        "db": db_check,
        "kafka": kafka_check,
        "model": model_check,
        "predictions": pred_check,
    }

    statuses = [c["status"] for c in checks.values()]
    if any(s == "error" for s in statuses):
        overall = "unhealthy"
    elif any(s in ("stale", "unavailable") for s in statuses):
        overall = "degraded"
    else:
        overall = "ok"

    return {
        "overall_status": overall,
        "checks": checks,
        "checked_at": datetime.now(timezone.utc).isoformat(),
    }
```

```python
    status = "ok"
    if settings.DATABASE_URL and pool is None:

            redis_status = "connected"
        except Exception:
            redis_status = "error"

    return HealthResponse(
        service=settings.SERVICE_NAME,

        db_pool=PoolStatus(**pool) if pool else None,
        redis_status=redis_status,
```
