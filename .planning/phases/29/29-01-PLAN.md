```typescript
/* ── Drift page types (Phase 29) ─────────────────── */

/** Parsed active model info for the drift page header card. */
export interface ActiveModelInfo {
  modelName: string;
  scalerVariant: string;
  version: number;
  trainedDate: string;          // ISO date from saved_at
  isActive: boolean;
  oosRmse: number | null;
  oosMae: number | null;
  oosDirectionalAccuracy: number | null;
}

/** A single data point for the rolling performance time series. */
export interface RollingPerformancePoint {
  date: string;
  rmse: number | null;
  mae: number | null;
  directionalAccuracy: number | null;
}

/** Retrain event summary for the status panel. */
export interface RetrainStatus {
  lastRetrainDate: string | null;
  isRetraining: boolean;
  oldModel: { name: string; rmse: number; mae: number } | null;
  newModel: { name: string; rmse: number; mae: number } | null;
  improvementPct: number | null;    // percentage improvement in RMSE
}

/** A single feature's distribution comparison (training vs recent). */
export interface FeatureDistribution {
  feature: string;
  trainingBins: { bin: string; count: number }[];
  recentBins: { bin: string; count: number }[];
  ksStat: number | null;
  psiValue: number | null;
  isDrifted: boolean;
}

/** Aggregated drift page data for the mock generator. */
export interface DriftPageData {
  activeModel: ActiveModelInfo | null;
  events: DriftEventEntry[];
  rollingPerformance: RollingPerformancePoint[];
  retrainStatus: RetrainStatus;
  featureDistributions: FeatureDistribution[];
}
```

    - prediction_drift: `{ baseline_rmse: 0.022, recent_rmse: 0.038, threshold: 0.033, error_ratio: 1.73 }`
    - concept_drift: `{ historical_rmse: 0.021, recent_rmse: 0.035, degradation_ratio: 1.67, threshold: 1.3 }`
  - Timestamps as ISO strings, spaced 1–3 days apart

  - RMSE: starts ~0.021, drifts to ~0.035

  - MAE: starts ~0.017, drifts to ~0.028
  - Directional Accuracy: starts ~0.63, drifts down to ~0.55
  - Add small random noise to each point

  - 10 histogram bins for recent data (shifted distribution for drifted features)
  - KS stat, PSI, and isDrifted flag (3 of 6 drifted)

Use deterministic values (no Math.random) so the mock data is stable across renders.
