```python
"""Prometheus metrics for the Kafka consumer service."""

from prometheus_client import Counter, Gauge, Histogram, start_http_server

messages_consumed_total = Counter(
    "messages_consumed_total",
    "Total Kafka messages consumed",
    ["topic"],
)

batch_write_duration_seconds = Histogram(
    "batch_write_duration_seconds",
    "Duration of batch write to PostgreSQL in seconds",
    ["table"],
    buckets=(0.005, 0.01, 0.025, 0.05, 0.1, 0.25, 0.5, 1.0, 2.5, 5.0, 10.0, 30.0),
)

consumer_lag = Gauge(
    "consumer_lag",
    "Kafka consumer lag (messages behind high watermark)",
    ["topic", "partition"],
)


def start_metrics_server(port: int = 9090) -> None:
    """Start a Prometheus metrics HTTP server on a daemon thread."""
    start_http_server(port)
```

Notes:
- start_http_server runs in a daemon thread by default — non-blocking
- Buckets for batch_write cover 5ms → 30s (DB batch writes can be slow under load)
- consumer_lag is a Gauge because it can go up and down

3. Add a helper function for lag tracking:
```python
def _update_consumer_lag(kafka_consumer) -> None:
    """Update consumer_lag gauge from current partition assignments."""
    try:
        partitions = kafka_consumer.assignment()
        for tp in partitions:
            low, high = kafka_consumer.get_watermark_offsets(tp)
            committed = kafka_consumer.committed([tp])
            offset = committed[0].offset if committed[0] and committed[0].offset >= 0 else 0
            lag = max(0, high - offset)
            consumer_lag.labels(topic=tp.topic, partition=str(tp.partition)).set(lag)
    except Exception:
        pass  # Partition not yet assigned or broker unavailable
```

```python
if processor.should_flush():

    consumer.commit(asynchronous=False)
```

```python
"""Tests for Kafka consumer Prometheus metrics."""

from unittest.mock import MagicMock, patch

import pytest
from prometheus_client import REGISTRY


@pytest.fixture(autouse=True)
def _reset_registry():
    """Allow tests to run independently by collecting before/after."""
    yield


def test_messages_consumed_counter_increments():
    from consumer.metrics import messages_consumed_total

    before = REGISTRY.get_sample_value(
        "messages_consumed_total", {"topic": "test-topic"}
    ) or 0.0

    messages_consumed_total.labels(topic="test-topic").inc()

    after = REGISTRY.get_sample_value(
        "messages_consumed_total", {"topic": "test-topic"}
    )
    assert after == before + 1.0


def test_batch_write_duration_observes():
    from consumer.metrics import batch_write_duration_seconds

    batch_write_duration_seconds.labels(table="ohlcv_daily").observe(0.123)

    count = REGISTRY.get_sample_value(
        "batch_write_duration_seconds_count", {"table": "ohlcv_daily"}
    )
    assert count is not None and count >= 1


@patch("consumer.metrics.start_http_server")
def test_start_metrics_server_calls_http_server(mock_http):
    from consumer.metrics import start_metrics_server

    start_metrics_server(9999)
    mock_http.assert_called_once_with(9999)
```
