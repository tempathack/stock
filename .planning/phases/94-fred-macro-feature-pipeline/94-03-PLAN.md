def get_fred_macro_features(self) -> dict:
    """Read the latest FRED macro row directly from feast_fred_macro.

    Bypasses Feast online store — fred_macro_fv uses entities=[] which is
    incompatible with the ticker-keyed get_online_features() call.

    Returns:

        recent feast_fred_macro row, or {} if the table is empty/unreachable.
    """
    import psycopg2
    import psycopg2.extras
    try:
        conn = psycopg2.connect(self._db_url)  # use whatever DB URL attr exists in the class
        try:
            with conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor) as cur:
                cur.execute(
                    "SELECT * FROM feast_fred_macro ORDER BY timestamp DESC LIMIT 1"

                return {k: v for k, v in dict(row).items() if k not in exclude}
        finally:
            conn.close()
    except Exception as exc:
        logger.warning("get_fred_macro_features: DB lookup failed: %s", exc)
        return {}
