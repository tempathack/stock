      - orgId: 1
        name: loki-log-alerts

          - uid: loki-high-error-log-rate

            condition: C
            data:
              - refId: A
                queryType: range
                relativeTimeRange:
                  from: 300

                  expr: 'sum(rate({namespace=~".+"} |= "ERROR" [5m]))'

                    - evaluator:

                      query:
                        params: [A]
            noDataState: NoData
            execErrState: Error
            for: 5m
            labels:
              severity: warning
              team: platform
            annotations:
              summary: "High ERROR log rate across namespaces"
