```python
"""Database connection and session management."""

from __future__ import annotations

import logging
from contextlib import contextmanager
from typing import Generator

logger = logging.getLogger(__name__)


@contextmanager
def get_db_connection(db_url: str | None) -> Generator:
    """Yield a psycopg2 connection, or None if unavailable.

    Usage::

        with get_db_connection(settings.DATABASE_URL) as conn:
            if conn is None:
                return fallback_result
            with conn.cursor() as cur:
                cur.execute(query)
    """
    if not db_url:
        yield None
        return

    try:
        import psycopg2
    except ImportError:
        logger.warning("psycopg2 not installed — DB features disabled")
        yield None
        return

    conn = None
    try:
        conn = psycopg2.connect(db_url)
        yield conn
    except Exception:
        logger.exception("Failed to connect to database")
        yield None
    finally:
        if conn is not None:
            try:
                conn.close()
            except Exception:
                pass
```

Key points:
- Context manager yields `None` when DB is unavailable (psycopg2 missing, bad URL, connection error)
- Callers always check `if conn is None` before using
- Matches the established pattern — no new dependencies, no SQLAlchemy (that's Phase 40)

def load_model_comparison_from_db(db_url: str | None) -> list[dict] | None:
    """Query model_registry table for model comparison data.

    Returns list of dicts matching ModelComparisonEntry shape, sorted by OOS RMSE.
    Returns None if DB is unavailable (caller should fall back to file-based).
    """
    from app.models.database import get_db_connection

    with get_db_connection(db_url) as conn:
        if conn is None:
            return None

        query = """
            SELECT model_name, version, metrics_json, trained_at, is_active
            FROM model_registry
            ORDER BY (metrics_json->>'oos_rmse')::numeric ASC NULLS LAST
        """
        try:
            with conn.cursor() as cur:
                cur.execute(query)
                cols = [desc[0] for desc in cur.description]
                rows = cur.fetchall()
        except Exception:
            logger.exception("Failed to query model_registry")
            return None

    entries = []
    for row in rows:
        raw = dict(zip(cols, row))
        metrics = raw.get("metrics_json") or {}
        entries.append({
            "model_name": raw.get("model_name", "unknown"),
            "scaler_variant": metrics.get("scaler_variant", "unknown"),
            "version": raw.get("version"),
            "is_winner": metrics.get("is_winner", False),
            "is_active": raw.get("is_active", False),
            "oos_metrics": {
                k: v for k, v in metrics.items()
                if k.startswith("oos_")
            },
            "fold_stability": metrics.get("fold_stability"),
            "best_params": metrics.get("best_params", {}),
            "saved_at": raw["trained_at"].isoformat() if raw.get("trained_at") else None,
        })
    return entries
```

Key points:
- Returns `None` (not empty list) when DB unavailable — the caller distinguishes "DB unavailable → fallback" from "DB available, no models → empty list"
- Extracts structured fields from metrics_json JSONB (psycopg2 auto-parses JSONB → dict)
- Output shape matches what `routers/models.py` already expects from `load_model_comparison()`
- Parameterized query (no user input in this query, but pattern is safe)

def load_drift_events_from_db(
    db_url: str | None,
    n: int = 100,
) -> list[dict] | None:
    """Query drift_logs table for recent drift events.

    Returns list of dicts matching DriftEventEntry shape, newest first.
    Returns None if DB is unavailable.
    """
    from app.models.database import get_db_connection

    with get_db_connection(db_url) as conn:
        if conn is None:
            return None

        query = """
            SELECT drift_type, severity, details_json, detected_at
            FROM drift_logs
            ORDER BY detected_at DESC
            LIMIT %s
        """
        try:
            with conn.cursor() as cur:
                cur.execute(query, (n,))
                cols = [desc[0] for desc in cur.description]
                rows = cur.fetchall()
        except Exception:
            logger.exception("Failed to query drift_logs")
            return None

    events = []
    for row in rows:
        raw = dict(zip(cols, row))
        details = raw.get("details_json") or {}
        events.append({
            "drift_type": raw.get("drift_type", "unknown"),
            "is_drifted": details.get("is_drifted", raw.get("severity", "none") != "none"),
            "severity": raw.get("severity", "none"),
            "details": details,
            "timestamp": raw["detected_at"].isoformat() if raw.get("detected_at") else None,
            "features_affected": [
                k for k, v in details.get("per_feature", {}).items()
                if v.get("drifted", False)
            ],
        })
    return events
```

Key points:
- Uses parameterized `%s` for the LIMIT value
- Derives `is_drifted` from details or severity (drift_logs table doesn't have an explicit is_drifted column)
- Derives `features_affected` from `per_feature` dict in details_json
- Output shape matches DriftEventEntry schema

def get_live_prediction(
    ticker: str,
    db_url: str | None = None,
    serving_dir: str = "/models/active",
) -> dict | None:
    """Run live inference for a single ticker using the active model pipeline.

    Flow:
    1. Load pipeline.pkl from serving_dir
    2. Fetch recent OHLCV data from DB (250 rows for indicator warmup)
    3. Compute all technical indicators
    4. Run pipeline.predict() on the latest feature row
    5. Return PredictionResponse-shaped dict

    Returns None if any step fails (model not found, DB unavailable, etc.).
    """
    import pickle
    from datetime import date
    from pathlib import Path

    import numpy as np
    import pandas as pd

    ticker = ticker.upper()
    pipeline_path = Path(serving_dir) / "pipeline.pkl"
    features_path = Path(serving_dir) / "features.json"
    metadata_path = Path(serving_dir) / "metadata.json"

    # Step 1: Load the active model pipeline
    if not pipeline_path.exists():
        logger.warning("No active model pipeline at %s", pipeline_path)
        return None

    try:
        with open(pipeline_path, "rb") as f:
            pipeline = pickle.load(f)  # noqa: S301
    except Exception:
        logger.exception("Failed to load pipeline from %s", pipeline_path)
        return None

    # Load feature names (required to align columns)
    feature_names = None
    if features_path.exists():
        with open(features_path) as f:
            feature_names = json.load(f)

    # Load model metadata for response
    model_name = "unknown"
    if metadata_path.exists():
        with open(metadata_path) as f:
            meta = json.load(f)
            model_name = meta.get("model_name", "unknown")
            if meta.get("scaler_variant"):
                model_name = f"{meta['model_name']}_{meta['scaler_variant']}"

    # Step 2: Fetch OHLCV data from DB
    ohlcv_df = _load_ohlcv_for_inference(ticker, db_url, lookback=250)
    if ohlcv_df is None or len(ohlcv_df) < 50:
        logger.warning(
            "Insufficient OHLCV data for %s (%d rows)",
            ticker,
            len(ohlcv_df) if ohlcv_df is not None else 0,
        )
        return None

    # Step 3: Compute features
    try:
        from ml.features.indicators import compute_all_indicators

        enriched = compute_all_indicators(ohlcv_df.copy())
    except Exception:
        logger.exception("Failed to compute indicators for %s", ticker)
        return None

    # Step 4: Align features and predict
    try:
        if feature_names:
            # Only keep columns the model was trained on
            available = [c for c in feature_names if c in enriched.columns]
            if len(available) < len(feature_names) * 0.8:
                logger.warning(
                    "Too many missing features for %s: %d/%d available",
                    ticker,
                    len(available),
                    len(feature_names),
                )
                return None
            X = enriched[available].iloc[[-1]]
            # Fill missing columns with 0
            for col in feature_names:
                if col not in X.columns:
                    X[col] = 0.0
            X = X[feature_names]
        else:
            # No feature list — drop non-numeric and use all
            X = enriched.select_dtypes(include=[np.number]).iloc[[-1]]

        X = X.replace([np.inf, -np.inf], np.nan).fillna(0.0)
        predicted_price = float(pipeline.predict(X)[0])
    except Exception:
        logger.exception("pipeline.predict() failed for %s", ticker)
        return None

    # Step 5: Build response
    last_close = float(ohlcv_df["close"].iloc[-1])
    confidence = max(0.0, min(1.0, 1.0 - abs(predicted_price - last_close) / last_close))

    return {
        "ticker": ticker,
        "prediction_date": str(date.today()),
        "predicted_date": str(date.today()),
        "predicted_price": round(predicted_price, 4),
        "model_name": model_name,
        "confidence": round(confidence, 4),
    }


def _load_ohlcv_for_inference(
    ticker: str,
    db_url: str | None,
    lookback: int = 250,
) -> "pd.DataFrame | None":
    """Load recent OHLCV rows for a single ticker from PostgreSQL."""
    if not db_url:
        return None

    from app.models.database import get_db_connection

    query = """
        SELECT date, open, high, low, close, adj_close, volume, vwap
        FROM ohlcv_daily
        WHERE ticker = %s
        ORDER BY date DESC
        LIMIT %s
    """

    with get_db_connection(db_url) as conn:
        if conn is None:
            return None
        try:
            import pandas as pd

            df = pd.read_sql_query(query, conn, params=(ticker.upper(), lookback))
            if df.empty:
                return None
            df = df.sort_values("date").reset_index(drop=True)
            return df
        except Exception:
            logger.exception("Failed to load OHLCV for %s", ticker)
            return None
```

Key points:
- Uses `pickle.load` with the serving dir pipeline (same pattern as ModelRegistry.load_model)
- Aligns feature columns to the model's expected feature list from features.json
- Confidence is a simple price-proximity heuristic (how close prediction is to last close)
- Falls back with None at every step — caller handles fallback
- `_load_ohlcv_for_inference` is a private helper, separate from market_service's version

def get_bulk_live_predictions(
    tickers: list[str],
    db_url: str | None = None,
    serving_dir: str = "/models/active",
) -> list[dict] | None:
    """Run live inference for all tickers.

    Returns list of prediction dicts, or None if live inference is
    completely unavailable (no pipeline, no DB).
    Skips individual ticker failures.
    """
    from pathlib import Path

    pipeline_path = Path(serving_dir) / "pipeline.pkl"
    if not pipeline_path.exists() or not db_url:
        return None

    results = []
    for ticker in tickers:
        pred = get_live_prediction(
            ticker=ticker,
            db_url=db_url,
            serving_dir=serving_dir,
        )
        if pred is not None:
            results.append(pred)

    return results if results else None
```

Key points:
- Quick-fail check before iterating (no pipeline or no DB → return None immediately)
- Skips individual ticker failures but collects successes
- Returns None if all tickers fail (caller falls back to cached predictions)

## Design Decisions

- **Return None vs empty list:** DB functions return `None` when the DB is unavailable (meaning "try fallback"), vs `[]` when DB is available but has no data (meaning "no results"). This lets the router distinguish the two cases.
- **No SQLAlchemy:** Phase 40 introduces async SQLAlchemy connection pooling. Phase 31 stays with psycopg2 to match the existing market_service.py pattern and avoid premature optimization.
- **Confidence heuristic:** Simple price-proximity confidence (1 - |delta|/price). Not a proper uncertainty estimate, but matches the existing schema. Phase 42 (ensemble stacking) will provide better confidence via ensemble variance.
- **Feature alignment:** The pipeline was trained on specific features stored in features.json. The inference code must align columns exactly, filling missing ones with 0.0. This is more robust than re-running the full training feature pipeline.
- **Private OHLCV loader:** `_load_ohlcv_for_inference` is separate from `market_service._load_ohlcv_from_db` because inference needs sorted ascending + may need different column handling. Consolidation can happen in Phase 40.
