```python
def load_rolling_performance_from_db(
    db_url: str | None,
    days: int = 30,
) -> dict | None:
    """Query rolling prediction performance for the active model.

    Joins predictions → ohlcv_daily to compare predicted_price vs actual close.
    Returns dict with 'entries' (per-date metrics) and 'model_name', or None if DB unavailable.
    """
    from app.models.database import get_db_connection

    with get_db_connection(db_url) as conn:
        if conn is None:
            return None

        query = """
            WITH active AS (
                SELECT model_id, model_name
                FROM model_registry
                WHERE is_active = true
                LIMIT 1
            ),
            compared AS (
                SELECT
                    p.predicted_date,
                    p.predicted_price,
                    o.close AS actual_price,
                    a.model_name
                FROM predictions p
                JOIN active a ON p.model_id = a.model_id
                JOIN ohlcv_daily o ON p.ticker = o.ticker AND p.predicted_date = o.date
                WHERE p.predicted_date >= CURRENT_DATE - %s * INTERVAL '1 day'
            )
            SELECT
                predicted_date::text AS date,
                model_name,
                sqrt(avg(power(predicted_price - actual_price, 2))) AS rmse,
                avg(abs(predicted_price - actual_price)) AS mae,
                count(*) AS n_predictions
            FROM compared
            GROUP BY predicted_date, model_name
            ORDER BY predicted_date ASC
        """
        try:
            with conn.cursor() as cur:
                cur.execute(query, (days,))
                cols = [desc[0] for desc in cur.description]
                rows = cur.fetchall()
        except Exception:
            logger.exception("Failed to query rolling performance")
            return None

    if not rows:
        return {"entries": [], "model_name": None}

    entries = []
    model_name = None
    for row in rows:
        raw = dict(zip(cols, row))
        model_name = raw.get("model_name")
        entries.append({
            "date": raw["date"],
            "rmse": round(float(raw["rmse"]), 6) if raw.get("rmse") else None,
            "mae": round(float(raw["mae"]), 6) if raw.get("mae") else None,
            "directional_accuracy": None,  # Requires lag data; computed client-side or in Phase 44
            "n_predictions": raw.get("n_predictions", 0),
        })

    return {"entries": entries, "model_name": model_name}
```

Key points:
- CTE `active` finds the current active model
- CTE `compared` joins predictions to actual OHLCV closes by (ticker, predicted_date = date)
- Aggregates per-date RMSE and MAE across all tickers
- `directional_accuracy` is left as None here — computing it requires previous day's close which adds complexity; Phase 44 (Feature Store) will precompute this
- Parameterized `%s` for the days interval

```python
def get_retrain_status_from_db(db_url: str | None) -> dict | None:
    """Get the latest retrain status from model_registry.

    Returns current active model info plus previous model for comparison.
    Returns None if DB is unavailable.
    """
    from app.models.database import get_db_connection

    with get_db_connection(db_url) as conn:
        if conn is None:
            return None

        query = """
            SELECT model_name, version, metrics_json, trained_at, is_active
            FROM model_registry
            ORDER BY trained_at DESC
            LIMIT 2
        """
        try:
            with conn.cursor() as cur:
                cur.execute(query)
                cols = [desc[0] for desc in cur.description]
                rows = cur.fetchall()
        except Exception:
            logger.exception("Failed to query retrain status")
            return None

    if not rows:
        return {
            "model_name": None,
            "version": None,
            "trained_at": None,
            "is_active": False,
            "oos_metrics": {},
            "previous_model": None,
            "previous_trained_at": None,
        }

    current = dict(zip(cols, rows[0]))
    metrics = current.get("metrics_json") or {}
    result = {
        "model_name": current.get("model_name"),
        "version": current.get("version"),
        "trained_at": current["trained_at"].isoformat() if current.get("trained_at") else None,
        "is_active": current.get("is_active", False),
        "oos_metrics": {k: v for k, v in metrics.items() if k.startswith("oos_")},
        "previous_model": None,
        "previous_trained_at": None,
    }

    if len(rows) > 1:
        prev = dict(zip(cols, rows[1]))
        result["previous_model"] = prev.get("model_name")
        result["previous_trained_at"] = prev["trained_at"].isoformat() if prev.get("trained_at") else None

    return result
```

Key points:
- Queries the 2 most recent models by trained_at to provide current + previous
- Extracts oos_* metrics from metrics_json for the current model
- Previous model is included so the frontend can show "replaced X with Y" context

```python
@router.get("/drift/rolling-performance", response_model=RollingPerformanceResponse)
async def drift_rolling_performance(days: int = 30) -> RollingPerformanceResponse:
    """Return rolling prediction error metrics for the active model."""
    result = load_rolling_performance_from_db(
        db_url=settings.DATABASE_URL,
        days=days,
    )
    if result is None:
        return RollingPerformanceResponse(entries=[], count=0)

    entries = [RollingPerfEntry(**e) for e in result.get("entries", [])]
    return RollingPerformanceResponse(
        entries=entries,
        model_name=result.get("model_name"),
        period_days=days,
        count=len(entries),
    )


@router.get("/retrain-status", response_model=RetrainStatusResponse)
async def retrain_status() -> RetrainStatusResponse:
    """Return metadata about the most recent model training event."""
    result = get_retrain_status_from_db(db_url=settings.DATABASE_URL)
    if result is None:
        return RetrainStatusResponse()
    return RetrainStatusResponse(**result)
```

## Design Decisions

- **Rolling performance uses DB-only (no file fallback):** These are new endpoints with no pre-existing file-based data source. When DB is unavailable, they return empty responses rather than 404/500. The frontend will show "No data available" in the chart area.
- **directional_accuracy left as None:** Computing whether the predicted direction matched actual requires comparing to the previous day's close, which adds query complexity. The frontend drift page can compute this client-side from the raw prediction data, or Phase 44 (Feature Store) will precompute it.
- **Retrain status shows current + previous:** The frontend drift page (FDRFT-04) shows "Model X replaced Model Y at timestamp Z". Having both models in one response avoids a second API call.
- **Route ordering:** `/drift/rolling-performance` must come before `/drift` in the router to avoid FastAPI treating "rolling-performance" as a path parameter. Since models.py uses a `/models` prefix and the drift endpoint is at `/drift` (not `/{param}`), this isn't strictly necessary — but reordering is defensive best practice.
- **`days` query parameter:** Allows the frontend to request different time windows (7d, 30d, 90d) without separate endpoints.
