  <name>Task 2: Mock indicator series generator</name>
  <context>Generate realistic time series data for the detail chart when the API is unavailable. Produces 90 days of OHLCV + technical indicators matching the IndicatorValues shape.</context>

  <files>
    <file path="services/frontend/src/utils/mockIndicatorData.ts" action="create">Mock indicator series generator:

      ```typescript
      import type { IndicatorValues } from "@/api";

      // Seeded PRNG (reuse same pattern from mockModelData)
      function seededRandom(seed: string): () => number { ... }

      /**
       * Generate 90 days of mock indicator data for a given ticker.
       * Produces realistic-looking OHLCV + technical indicator values.
       */
      export function generateMockIndicatorSeries(ticker: string): IndicatorValues[] {
        const rng = seededRandom(ticker + "_indicators");
        const days = 90;
        let price = 100 + rng() * 400; // Base price 100-500

        return Array.from({ length: days }, (_, i) => {
          const date = new Date(Date.now() - (days - i) * 86400000)
            .toISOString().split("T")[0];

          // Random walk price
          price *= 1 + (rng() - 0.48) * 0.03;

          const close = Math.round(price * 100) / 100;
          const sma20 = close * (0.98 + rng() * 0.04);
          const sma50 = close * (0.96 + rng() * 0.08);
          const sma200 = close * (0.90 + rng() * 0.20);
          const ema12 = close * (0.99 + rng() * 0.02);
          const ema26 = close * (0.98 + rng() * 0.04);

          return {
            date,
            close,
            rsi_14: 30 + rng() * 40,            // 30-70 range
            macd_line: (rng() - 0.5) * 2,
            macd_signal: (rng() - 0.5) * 1.5,
            macd_histogram: (rng() - 0.5) * 1,
            stoch_k: 20 + rng() * 60,
            stoch_d: 20 + rng() * 60,
            sma_20: Math.round(sma20 * 100) / 100,
            sma_50: Math.round(sma50 * 100) / 100,
            sma_200: Math.round(sma200 * 100) / 100,
            ema_12: Math.round(ema12 * 100) / 100,
            ema_26: Math.round(ema26 * 100) / 100,
            adx: 15 + rng() * 35,
            bb_upper: Math.round((close * 1.04) * 100) / 100,
            bb_middle: Math.round(sma20 * 100) / 100,
            bb_lower: Math.round((close * 0.96) * 100) / 100,
            atr: Math.round((close * 0.015) * 100) / 100,
            rolling_volatility_21: 0.1 + rng() * 0.3,
            obv: Math.round(1000000 + rng() * 5000000),
            vwap: Math.round((close * (0.99 + rng() * 0.02)) * 100) / 100,
            volume_sma_20: Math.round(500000 + rng() * 2000000),
            ad_line: Math.round((rng() - 0.5) * 1000000),
            return_1d: (rng() - 0.5) * 4,
            return_5d: (rng() - 0.5) * 8,
            return_21d: (rng() - 0.5) * 15,
            log_return: (rng() - 0.5) * 0.04,
          };
        });
      }
      ```
