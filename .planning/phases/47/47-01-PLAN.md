Create `services/api/app/cache.py` with the following structure:

```python
"""Redis caching layer — async client, TTL constants, get/set/invalidate helpers."""

from __future__ import annotations

import json
import logging
from typing import Any

import redis.asyncio as aioredis

logger = logging.getLogger(__name__)

# --- TTL constants (seconds) ---
MARKET_OVERVIEW_TTL = 30
MARKET_INDICATORS_TTL = 30
BULK_PREDICTIONS_TTL = 60
SINGLE_PREDICTION_TTL = 60
MODEL_COMPARISON_TTL = 300
DRIFT_STATUS_TTL = 300
BACKTEST_TTL = 120
HORIZONS_TTL = 600

# --- Module-level singleton ---
_redis_client: aioredis.Redis | None = None


def init_redis(redis_url: str) -> None:
    global _redis_client
    _redis_client = aioredis.from_url(redis_url, decode_responses=True)
    logger.info("redis client initialized", extra={"redis_url": redis_url.split("@")[-1]})


async def close_redis() -> None:
    global _redis_client
    if _redis_client is not None:
        await _redis_client.close()
        _redis_client = None
        logger.info("redis client closed")


def get_redis() -> aioredis.Redis | None:
    return _redis_client


async def cache_get(key: str) -> dict | list | None:
    if _redis_client is None:
        return None
    try:
        value = await _redis_client.get(key)
        if value is not None:
            return json.loads(value)
    except aioredis.RedisError:
        logger.warning("redis GET failed", extra={"key": key}, exc_info=True)
    return None


async def cache_set(key: str, value: Any, ttl: int) -> None:
    if _redis_client is None:
        return
    try:
        await _redis_client.setex(key, ttl, json.dumps(value, default=str))
    except aioredis.RedisError:
        logger.warning("redis SET failed", extra={"key": key}, exc_info=True)


async def cache_delete(pattern: str) -> int:
    if _redis_client is None:
        return 0
    count = 0
    try:
        async for key in _redis_client.scan_iter(match=pattern):
            await _redis_client.delete(key)
            count += 1
    except aioredis.RedisError:
        logger.warning("redis DELETE failed", extra={"pattern": pattern}, exc_info=True)
    return count


async def invalidate_predictions() -> int:
    return await cache_delete("predict:*")


async def invalidate_models() -> int:
    return await cache_delete("models:*")


async def invalidate_all() -> int:
    return await cache_delete("*")


def build_key(*parts: str) -> str:
    return ":".join(parts)
```

Key design decisions:
- **Graceful degradation:** Every function is a no-op when `_redis_client is None` (Redis not configured)
- **Never raises:** All Redis errors are caught and logged as warnings — cache failures don't break API
- **JSON serialization:** `default=str` handles datetime objects safely
- **SCAN-based delete:** Safe for large keyspaces (no KEYS command)
- **build_key helper:** Consistent key format like `predict:AAPL:7`, `models:comparison`, `market:overview`
