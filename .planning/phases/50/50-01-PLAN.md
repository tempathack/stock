/**
 * Generate a CSV file from headers + rows and trigger browser download.
 */
export function exportToCsv(
  filename: string,
  headers: string[],
  rows: string[][],
): void {
  const escape = (val: string): string => {
    if (/[",\n\r]/.test(val)) {
      return `"${val.replace(/"/g, '""')}"`;
    }
    return val;
  };

  const csvLines = [
    headers.map(escape).join(","),
    ...rows.map((row) => row.map(escape).join(",")),
  ];

  const csvString = "\uFEFF" + csvLines.join("\r\n");
  const blob = new Blob([csvString], { type: "text/csv;charset=utf-8;" });
  const url = URL.createObjectURL(blob);

  const link = document.createElement("a");
  link.href = url;
  link.download = filename;
  link.style.display = "none";
  document.body.appendChild(link);
  link.click();
  document.body.removeChild(link);
  URL.revokeObjectURL(url);
}
```

Key points:
- BOM prefix `\uFEFF` ensures Excel opens UTF-8 correctly
- Values with commas, quotes, or newlines are escaped per RFC 4180
- Hidden `<a>` element triggers download then cleans up

import { jsPDF } from "jspdf";
import autoTable from "jspdf-autotable";

/**
 * Generate a PDF with a title, optional metadata lines, and a data table.
 */
export function exportTableToPdf(
  filename: string,
  title: string,
  headers: string[],
  rows: string[][],
  metadata?: string[],
): void {
  const doc = new jsPDF({ orientation: "landscape" });

  // Title
  doc.setFontSize(16);
  doc.setTextColor(224, 224, 224);
  doc.setFillColor(26, 26, 46); // bg-primary
  doc.rect(0, 0, doc.internal.pageSize.getWidth(), doc.internal.pageSize.getHeight(), "F");
  doc.text(title, 14, 18);

  // Metadata
  let startY = 26;
  if (metadata?.length) {
    doc.setFontSize(9);
    doc.setTextColor(160, 160, 160);
    for (const line of metadata) {
      doc.text(line, 14, startY);
      startY += 5;
    }
    startY += 4;
  }

  // Table
  autoTable(doc, {
    head: [headers],
    body: rows,
    startY,
    theme: "grid",
    styles: {
      fillColor: [26, 26, 46],
      textColor: [224, 224, 224],
      lineColor: [42, 42, 74],
      lineWidth: 0.3,
      fontSize: 8,
    },
    headStyles: {
      fillColor: [15, 52, 96],
      textColor: [255, 255, 255],
      fontStyle: "bold",
      fontSize: 8,
    },
    alternateRowStyles: {
      fillColor: [22, 33, 62],
    },
  });

  doc.save(filename);
}
```

Key points:
- Landscape orientation to fit wide data tables
- Dark theme matching the frontend Bloomberg style
- jspdf-autotable handles pagination, column widths, and page breaks automatically

export default function ExportButtons({
  onExportCsv,
  onExportPdf,
  disabled = false,
  className = "",
}: ExportButtonsProps) {
  const base =
    "inline-flex items-center gap-1.5 rounded-md border border-border bg-bg-card px-3 py-1.5 text-xs font-medium transition-colors";
  const enabled = "text-text-secondary hover:bg-border hover:text-text-primary";
  const off = "cursor-not-allowed opacity-40 text-text-secondary";

  return (
    <div className={`flex items-center gap-2 ${className}`}>
      <button
        onClick={onExportCsv}
        disabled={disabled}
        className={`${base} ${disabled ? off : enabled}`}
        title="Export as CSV"
      >
        📄 CSV
      </button>
      <button
        onClick={onExportPdf}
        disabled={disabled}
        className={`${base} ${disabled ? off : enabled}`}
        title="Export as PDF"
      >
        📑 PDF
      </button>
    </div>
  );
}
