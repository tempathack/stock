**Create `k8s/processing/kafka-connect-debezium.yaml`:**

```yaml
apiVersion: kafka.strimzi.io/v1beta2
kind: KafkaConnect
metadata:
  name: debezium-connect
  namespace: processing
  labels:
    app: debezium-connect
    app.kubernetes.io/part-of: stock-prediction-platform
  annotations:
    # REQUIRED: enables KafkaConnector CRs to be reconciled by the operator
    strimzi.io/use-connector-resources: "true"
spec:
  version: 3.7.0
  replicas: 1
  # pre-built image loaded via: minikube image load debezium-connect-custom:latest
  image: debezium-connect-custom:latest
  bootstrapServers: kafka-kafka-bootstrap.storage.svc.cluster.local:9092
  config:
    group.id: debezium-connect-cluster
    offset.storage.topic: debezium-connect-offsets
    config.storage.topic: debezium-connect-configs
    status.storage.topic: debezium-connect-status
    config.storage.replication.factor: 1
    offset.storage.replication.factor: 1
    status.storage.replication.factor: 1
    # Converters: use JSON without schemas for simplicity
    key.converter: org.apache.kafka.connect.json.JsonConverter
    value.converter: org.apache.kafka.connect.json.JsonConverter
    key.converter.schemas.enable: "false"
    value.converter.schemas.enable: "false"
  resources:
    requests:
      memory: "512Mi"
      cpu: "500m"
    limits:
      memory: "1Gi"
      cpu: "1"
  template:
    pod:
      metadata:
        labels:
          app: debezium-connect
      spec:
        imagePullPolicy: Never
```

**Create `k8s/processing/kafka-connect-connector-pg.yaml`:**

```yaml
apiVersion: kafka.strimzi.io/v1beta2
kind: KafkaConnector
metadata:
  name: debezium-postgres-source
  namespace: processing
  labels:
    strimzi.io/cluster: debezium-connect
spec:
  class: io.debezium.connector.postgresql.PostgresConnector
  tasksMax: 1
  config:
    database.hostname: postgresql.storage.svc.cluster.local
    database.port: "5432"
    database.user: stockuser
    database.password: stockpassword
    database.dbname: stockdb
    database.server.name: debezium
    topic.prefix: debezium
    plugin.name: pgoutput
    publication.name: dbz_publication
    slot.name: debezium_slot
    snapshot.mode: initial
    table.include.list: "public.predictions,public.drift_logs,public.model_registry"
    heartbeat.interval.ms: "10000"
    # Flatten CDC envelope: removes before/after wrapper, adds __op and __ts_ms fields
    transforms: unwrap
    transforms.unwrap.type: io.debezium.transforms.ExtractNewRecordState
    transforms.unwrap.drop.tombstones: "false"
    transforms.unwrap.delete.handling.mode: rewrite
    transforms.unwrap.add.fields: "op,ts_ms"
```

**Create `k8s/processing/kafka-connect-connector-es.yaml`:**

```yaml
apiVersion: kafka.strimzi.io/v1beta2
kind: KafkaConnector
metadata:
  name: es-sink-connector
  namespace: processing
  labels:
    strimzi.io/cluster: debezium-connect
spec:
  class: io.confluent.connect.elasticsearch.ElasticsearchSinkConnector
  tasksMax: 1
  config:
    topics: "debezium.public.predictions,debezium.public.drift_logs,debezium.public.model_registry"
    connection.url: http://elasticsearch.storage.svc.cluster.local:9200
    type.name: _doc
    key.ignore: "false"
    schema.ignore: "true"
    behavior.on.null.values: delete
    batch.size: "100"
    linger.ms: "1000"
    index.name.format: "${topic}"
    # Key converter for ES Sink
    key.converter: org.apache.kafka.connect.storage.StringConverter
    value.converter: org.apache.kafka.connect.json.JsonConverter
    value.converter.schemas.enable: "false"
```
