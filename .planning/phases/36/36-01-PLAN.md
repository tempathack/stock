Create the actual secret manifest at `k8s/storage/secrets.yaml`:

```yaml
# IMPORTANT: This file contains real credentials. NEVER commit to version control.
# It is listed in .gitignore. Copy from secrets.yaml.example and fill in real base64 values.
#
# To encode a value:  echo -n 'yourpassword' | base64
# To decode a value:  echo 'base64string' | base64 -d

apiVersion: v1
kind: Secret
metadata:
  name: stock-platform-secrets
  namespace: storage
  labels:
    app.kubernetes.io/part-of: stock-prediction-platform
type: Opaque
data:
  # echo -n 'devpassword123' | base64 → ZGV2cGFzc3dvcmQxMjM=
  POSTGRES_PASSWORD: ZGV2cGFzc3dvcmQxMjM=
  # echo -n 'postgresql://stockuser:devpassword123@postgresql.storage.svc.cluster.local:5432/stockdb' | base64
  DATABASE_URL: cG9zdGdyZXNxbDovL3N0b2NrdXNlcjpkZXZwYXNzd29yZDEyM0Bwb3N0Z3Jlc3FsLnN0b3JhZ2Uuc3ZjLmNsdXN0ZXIubG9jYWw6NTQzMi9zdG9ja2Ri
```

Create the template at `k8s/storage/secrets.yaml.example`:

```yaml
# Template for stock-platform-secrets.
# Copy this file to secrets.yaml and replace base64 values with real credentials.
#
# To encode a value:  echo -n 'yourpassword' | base64
# To decode a value:  echo 'base64string' | base64 -d
#
# For production, use Sealed Secrets or an external secrets operator:
#   kubeseal --format yaml < secrets.yaml > sealed-secrets.yaml
#   kubectl apply -f sealed-secrets.yaml

apiVersion: v1
kind: Secret
metadata:
  name: stock-platform-secrets
  namespace: storage
  labels:
    app.kubernetes.io/part-of: stock-prediction-platform
type: Opaque
data:
  # echo -n 'CHANGEME' | base64 → Q0hBTkdFTUU=
  POSTGRES_PASSWORD: Q0hBTkdFTUU=
  # echo -n 'postgresql://stockuser:CHANGEME@postgresql.storage.svc.cluster.local:5432/stockdb' | base64
  DATABASE_URL: Q0hBTkdFTUU=
```
