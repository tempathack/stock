```python
"""Yahoo Finance data fetching service."""
from __future__ import annotations

import json
from datetime import datetime, timezone

import pandas as pd
import yfinance as yf
import requests
from tenacity import (
    retry,
    stop_after_attempt,
    wait_exponential,
    retry_if_exception_type,
)

from app.config import settings
from app.utils.logging import get_logger

logger = get_logger(__name__)

DEFAULT_TICKERS = [
    "AAPL", "MSFT", "GOOGL", "AMZN", "NVDA",
    "META", "TSLA", "BRK-B", "JPM", "JNJ",
    "V", "PG", "UNH", "HD", "MA",
    "BAC", "XOM", "PFE", "ABBV", "CVX",
]
```

**Class `YahooFinanceService`:**

- `__init__(self)`: Read `settings.TICKER_SYMBOLS`, split on comma, strip whitespace. Store as `self.tickers: list[str]`. If TICKER_SYMBOLS is empty string, fall back to `DEFAULT_TICKERS`.

- Property or attribute `tickers` exposing the ticker list.

- `fetch_intraday(self, tickers: list[str] | None = None) -> list[dict]`: For each ticker (defaults to self.tickers), call `_fetch_ticker_data(ticker, period="1d", interval="1m")`. Validate returned DataFrame. Collect valid records with `fetch_mode="intraday"`. Log valid/skip counts per ticker at INFO. Return all valid records.

- `fetch_historical(self, tickers: list[str] | None = None) -> list[dict]`: Same pattern but `period="5y"`, `interval="1d"`, `fetch_mode="historical"`.

- `_fetch_and_validate(self, ticker: str, period: str, interval: str, fetch_mode: str) -> list[dict]`: Shared logic for both modes. Calls `_fetch_ticker_data`, checks `df.empty`, validates, converts to records. Handle `tenacity.RetryError` (all retries exhausted) by logging ERROR and returning empty list.

- `validate_ohlcv(self, df: pd.DataFrame, ticker: str) -> tuple[list[dict], int]`: Row-by-row validation:
  1. Reject if any of `Open`, `High`, `Low`, `Close` is NaN (`pd.isna()`)
  2. Reject if `Volume` is NaN or < 0 (but Volume=0 is VALID)
  3. Reject if `High < Low`
  4. Log each rejection at WARNING level with `event="invalid_record"`, `ticker=ticker`, `timestamp=str(idx)`, `reason=<one of "null_ohlc", "invalid_volume", "high_lt_low">`
  5. Valid rows: build record dict with keys: `ticker`, `timestamp` (UTC ISO 8601), `open`, `high`, `low`, `close`, `volume` (cast to int), `fetch_mode`, `ingested_at` (UTC ISO 8601 now)
  Return `(valid_records, skip_count)`

- `_normalize_timestamp(self, idx_value) -> str`: Convert pandas Timestamp to UTC ISO 8601. If tz-aware: `tz_convert("UTC")`. If tz-naive: `tz_localize("UTC")`. Return `.isoformat()`.

**Module-level function with tenacity retry decorator:**
```python
@retry(
    stop=stop_after_attempt(3),
    wait=wait_exponential(multiplier=2, min=2, max=8),
    retry=retry_if_exception_type((requests.ConnectionError, requests.Timeout, requests.HTTPError)),
    reraise=True,
)
def _fetch_ticker_data(ticker: str, period: str, interval: str) -> pd.DataFrame:
    """Fetch OHLCV data for a single ticker with retry."""
    df = yf.download(ticker, period=period, interval=interval, progress=False)
    # Safety: handle MultiIndex columns (shouldn't happen in 0.2.38 for single ticker)
    if isinstance(df.columns, pd.MultiIndex):
        df.columns = df.columns.droplevel(1)
    return df
```

Key implementation details:
- Use `float(row["Open"])` etc. when building record dicts to ensure JSON-serializable floats (not numpy floats)
- Cast volume to `int(row["Volume"])` for clean JSON
- `ingested_at` uses `datetime.now(timezone.utc).isoformat()`
- Lowercase keys in output dict: `open`, `high`, `low`, `close`, `volume` (not `Open`, `High`)
