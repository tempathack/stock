```typescript
/* ── Rolling Performance & Retrain Status (Phase 32) ─── */

export interface RollingPerfEntry {
  date: string;
  rmse: number | null;
```

3. Add two new hook functions after `useModelDrift()`:

```typescript
export function useRollingPerformance(days = 30) {
  return useQuery({
    queryKey: queryKeys.rollingPerformance(days),
    queryFn: async () => {
      const { data } = await apiClient.get<RollingPerformanceResponse>(
        `/models/drift/rolling-performance`,
        { params: { days } },
      );
      return data;
    },
  });
}

export function useRetrainStatus() {
  return useQuery({
    queryKey: queryKeys.retrainStatus,
    queryFn: async () => {
      const { data } = await apiClient.get<RetrainStatusResponse>(
        "/models/retrain-status",
      );
      return data;
    },
  });
}
```

Key points:
- useRollingPerformance takes optional `days` param (default 30), forwarded as query parameter
- Both hooks follow the same pattern as existing hooks (useQuery + apiClient.get)
- Query keys include the endpoint path segments for natural invalidation
- No special retry/interval config — use QueryProvider defaults (1 retry, 30s stale)

Major refactor of Drift.tsx. Replace mock-as-primary with API-first loading pattern.

1. **Add new hook imports:**


2. **Add new queries in component body:**

```tsx
const rollingPerfQuery = useRollingPerformance();

```

3. **Update loading guard to include all queries:**

```tsx
// BEFORE:
if (driftQuery.isLoading && modelsQuery.isLoading) return <LoadingSpinner />;

// AFTER:
const isLoading = driftQuery.isLoading && modelsQuery.isLoading
  && rollingPerfQuery.isLoading && retrainQuery.isLoading;
if (isLoading) return <LoadingSpinner />;
```

Note: Using `&&` (all loading) matches existing behavior — show spinner only when ALL queries are still loading, so partial data can render as individual queries resolve.

4. **Wire rolling performance to API data with mock fallback:**

```tsx
// BEFORE:
const rollingPerformance = mockData.rollingPerformance;

// AFTER:
const rollingPerformance = useMemo(() => {
  if (rollingPerfQuery.data?.entries?.length) {
    return rollingPerfQuery.data.entries.map((e) => ({
      date: e.date,
      rmse: e.rmse,
      mae: e.mae,
      directionalAccuracy: e.directional_accuracy,
    }));
  }
  if (rollingPerfQuery.isError) return mockData.rollingPerformance;
  return [];
}, [rollingPerfQuery.data, rollingPerfQuery.isError, mockData]);
```

This maps the API's snake_case `RollingPerfEntry` to the component's camelCase `RollingPerformancePoint`.

5. **Wire retrain status to API data with mock fallback:**

```tsx
// BEFORE:
const retrainStatus = mockData.retrainStatus;

// AFTER:
const retrainStatus = useMemo<RetrainStatus>(() => {
  if (retrainQuery.data?.model_name) {
    const d = retrainQuery.data;
    return {
      lastRetrainDate: d.trained_at,
      isRetraining: false,
      oldModel: d.previous_model
        ? { name: d.previous_model, rmse: 0, mae: 0 }
        : null,
      newModel: d.model_name
        ? {
            name: d.model_name,
            rmse: (d.oos_metrics?.oos_rmse as number) ?? 0,
            mae: (d.oos_metrics?.oos_mae as number) ?? 0,
          }
        : null,
      improvementPct: null,
    };
  }
  if (retrainQuery.isError) return mockData.retrainStatus;
  return mockData.retrainStatus; // Default until API returns data
}, [retrainQuery.data, retrainQuery.isError, mockData]);
```

Note: The retrain status mapping is lossy because the backend doesn't provide old model metrics or improvementPct. Use null/0 for unavailable fields.
